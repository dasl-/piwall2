# broadcaster (now we need ffmpeg's `-re` flag because the network doesn't provide backpressure like the pipe output did)
yt_url="https://www.youtube.com/watch?v=hE2Ira-Cwxo" ; ffmpeg -re -i <(youtube-dl $yt_url -f 'bestvideo' -o -) -i <(youtube-dl $yt_url -f 'bestaudio' -o -) -c:v copy -c:a aac -f matroska udp://239.0.1.23:1234
